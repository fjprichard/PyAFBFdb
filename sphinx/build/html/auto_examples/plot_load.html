# Import a simulation protocol.
from afbfdb import protocol


# directory to save examples.
root_dir = "../data/"
data_dir = root_dir + "SimulationSet_001/"

# Load the protocol.
simu = protocol(data_dir)
# Show an example.
simu.ShowExample(5)
